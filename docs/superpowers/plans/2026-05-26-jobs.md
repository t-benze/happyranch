# Jobs Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rename the `scripts` module to `jobs` and extend the submission form with two policy flags (`review_required`, `persistent`), so agents can either request founder-approved one-shots or run autonomous long-running jobs.

**Architecture:** Single SQLite-backed module per org, one DB table (`jobs`), one runner subprocess pool, one Feishu notification kind, one CLI noun, one web feature. The two boolean flags on each submission drive all behavior differences — no mode enum, no daemon-side classification.

**Tech Stack:** Python 3.11+ / FastAPI / SQLite (per-org DB) / asyncio subprocess / React + TypeScript / Tailwind / TanStack Query / `lark-oapi` (Feishu).

**Reference spec:** `docs/superpowers/specs/2026-05-26-jobs-design.md`

```python
    _seed_legacy_scripts_db(db_path)

    # Initializing Database against the legacy file should run the migration.
    Database(str(db_path))

    conn = sqlite3.connect(str(db_path))
    try:
        # 1. Table renamed
        tables = [r[0] for r in conn.execute(
            "SELECT name FROM sqlite_master WHERE type='table'")]
        assert "jobs" in tables
        assert "script_requests" not in tables

        # 2. IDs rewritten
        ids = sorted(r[0] for r in conn.execute("SELECT id FROM jobs"))
        assert ids == ["JOB-001", "JOB-002", "JOB-003"]

        # 3. Output paths rewritten

        assert paths["JOB-003"] == "/runtime/orgs/sample/jobs/JOB-003.out"

        assert row == ("failed", "daemon_crash")

        cols = {r[1]: r for r in conn.execute("PRAGMA table_info(jobs)")}
        assert "review_required" in cols
        assert "persistent" in cols

        assert "reason" in cols

            "SELECT review_required, persistent FROM jobs"))
        for rr, pers in legacy_flags:
            assert rr == 1
            assert pers == 0

        runtimes = dict(conn.execute(
            "SELECT id, max_runtime_seconds FROM jobs"))
        assert runtimes["JOB-001"] == 300
        assert runtimes["JOB-003"] == 600

        assert "timeout_seconds" not in cols
```

```python

            # Detect pre-rename state: script_requests exists AND jobs does not.
            cur = conn.execute(
                "SELECT name FROM sqlite_master "
                "WHERE type='table' AND name IN ('script_requests','jobs')"
            )
            tables = {r[0] for r in cur.fetchall()}
            if "script_requests" in tables and "jobs" not in tables:
                conn.executescript("""
                    BEGIN;
                    ALTER TABLE script_requests RENAME TO jobs;

                    ALTER TABLE jobs ADD COLUMN review_required INTEGER NOT NULL DEFAULT 0;
                    ALTER TABLE jobs ADD COLUMN persistent INTEGER NOT NULL DEFAULT 0;
                    ALTER TABLE jobs ADD COLUMN max_output_bytes INTEGER NOT NULL DEFAULT 52428800;
                    ALTER TABLE jobs ADD COLUMN stdout_bytes INTEGER;
                    ALTER TABLE jobs ADD COLUMN stderr_bytes INTEGER;
                    ALTER TABLE jobs ADD COLUMN reason TEXT;
                    ALTER TABLE jobs ADD COLUMN max_runtime_seconds INTEGER;

                    UPDATE jobs SET max_runtime_seconds = timeout_seconds
                     WHERE timeout_seconds IS NOT NULL;
                    ALTER TABLE jobs DROP COLUMN timeout_seconds;

                    -- Backfill legacy rows to the semantics they had under SR
                    UPDATE jobs SET review_required = 1 WHERE review_required = 0;

                    -- Force-fail any orphaned 'running' rows (daemon has exited)
                    UPDATE jobs
                       SET status = 'failed',
                           reason = 'daemon_crash',
                           finished_at = COALESCE(finished_at, started_at, created_at)
                     WHERE status = 'running';

                    -- ID rewrite: SR-NNN → JOB-NNN
                    UPDATE jobs SET id = 'JOB-' || SUBSTR(id, 4) WHERE id LIKE 'SR-%';

                    -- Path rewrite: .../scripts/SR-NNN.* → .../jobs/JOB-NNN.*
                    UPDATE jobs
                       SET stdout_path = REPLACE(REPLACE(stdout_path, '/scripts/SR-', '/jobs/JOB-'),
                                                 '/scripts/', '/jobs/')
                     WHERE stdout_path IS NOT NULL;
                    UPDATE jobs
                       SET stderr_path = REPLACE(REPLACE(stderr_path, '/scripts/SR-', '/jobs/JOB-'),
                                                 '/scripts/', '/jobs/')
                     WHERE stderr_path IS NOT NULL;

                    -- Cross-reference rewrites
                    UPDATE escalation_notifications
                       SET task_id = 'JOB-' || SUBSTR(task_id, 4)
                     WHERE kind = 'script_request' AND task_id LIKE 'SR-%';

                    UPDATE escalation_notifications
                       SET kind = 'job_request'
                     WHERE kind = 'script_request';

                    -- Audit kind + payload rewrites
                    UPDATE audit_log
                       SET kind = 'job_' || SUBSTR(kind, 8)
                     WHERE kind LIKE 'script_%';

                    UPDATE audit_log
                       SET payload_json = REPLACE(payload_json, '"script_id"', '"job_id"')
                     WHERE payload_json LIKE '%"script_id"%';

                    UPDATE audit_log
                       SET payload_json = REPLACE(payload_json, '"SR-', '"JOB-')
                     WHERE payload_json LIKE '%"SR-%';

                    -- Index rename
                    DROP INDEX IF EXISTS idx_script_requests_task;
                    DROP INDEX IF EXISTS idx_script_requests_agent;
                    DROP INDEX IF EXISTS idx_script_requests_status;
                    DROP INDEX IF EXISTS idx_script_requests_created_at;
                    CREATE INDEX IF NOT EXISTS jobs_task_id_idx ON jobs(task_id);
                    CREATE INDEX IF NOT EXISTS jobs_status_idx  ON jobs(status);
```

```python
from pathlib import Path

def migrate_filesystem_layout(org_root: Path | str) -> None:
    """Rename <org_root>/scripts/ → <org_root>/jobs/ and each SR-* file to JOB-*.

    Idempotent: a no-op if jobs/ already exists OR if scripts/ doesn't exist.
    Called once per org at daemon startup, before any runner uses the directory.
    """
    org_root = Path(org_root)
    scripts_dir = org_root / "scripts"
    jobs_dir = org_root / "jobs"

    if jobs_dir.exists():
        return  # already migrated (or fresh install)
    if not scripts_dir.exists():
        return  # nothing to migrate

    scripts_dir.rename(jobs_dir)
    for entry in jobs_dir.iterdir():
        if entry.name.startswith("SR-"):
            entry.rename(jobs_dir / ("JOB-" + entry.name[len("SR-"):]))
```

In `src/daemon/jobs_runner.py`:

```python
async def terminate_jobs_for_task(
    task_id: str,
    *,
    inflight_to_task: dict[str, str] | None = None,
    grace_seconds: float = 5.0,
) -> list[str]:
    """Kill every in-flight job whose owning task_id matches.

    `inflight_to_task` is a mapping JOB-NNN → TASK-NNN provided by the caller
    (usually built by scanning the DB). The test path passes it directly; the
    daemon path resolves it via the per-org DB before calling.

    Returns the list of JOB-NNN ids that were terminated.
    """
    inflight_to_task = inflight_to_task or {}
    targets = [
        (job_id, proc)
        for job_id, proc in _INFLIGHT.items()
        if inflight_to_task.get(job_id) == task_id
    ]
    if not targets:
        return []
    for job_id, proc in targets:
        try:
            os.killpg(proc.pid, signal.SIGTERM)
        except ProcessLookupError:
            pass
    # The runner sets reason on natural exit; for task_ended we mark via a
    # module-level dict so the runner reads it during its post-kill bookkeeping.
    for job_id, _ in targets:
        _KILL_REASON_OVERRIDE[job_id] = "task_ended"
    await asyncio.sleep(grace_seconds)
    for job_id, proc in targets:
        if proc.returncode is None:
            try:
                os.killpg(proc.pid, signal.SIGKILL)
            except ProcessLookupError:
                pass
    return [job_id for job_id, _ in targets]
```

```python
    job_id: str,

    task_id: str | None = None,
```

```python
    bus = org.event_bus.subscribe(channel=f"job:{job_id}")
```
