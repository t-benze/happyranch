```python
def org_age_days(db, *, now: datetime | None = None) -> int:
    """Days between the earliest audit_log row and `now`. Empty DB → 0."""
    row = db.fetchone("SELECT MIN(timestamp) AS first_ts FROM audit_log")
    if not row or not row["first_ts"]:
        return 0
    first = datetime.fromisoformat(row["first_ts"])
    moment = now or datetime.now(timezone.utc)
    if first.tzinfo is None:
        first = first.replace(tzinfo=timezone.utc)
    if moment.tzinfo is None:
        moment = moment.replace(tzinfo=timezone.utc)
    return max(0, (moment - first).days)
```

```python
def _local_midnight(now: datetime) -> datetime:
    return now.replace(hour=0, minute=0, second=0, microsecond=0)


def spend_today(db, *, now: datetime) -> float:
    """Sum cost_usd in session_token_usage from local midnight to now."""
    midnight = _local_midnight(now).isoformat()
    row = db.fetchone(
        "SELECT COALESCE(SUM(cost_usd), 0) AS total "
        "FROM session_token_usage WHERE created_at >= ?",
        (midnight,),
    )
    return float(row["total"]) if row else 0.0
```

```python
def narrative_counts_today(db, *, now: datetime, kb_store) -> NarrativeCounts:
    """Today = local midnight to now. Aggregates over tasks + audit_log +
    session_token_usage + the KB store."""
    midnight = _local_midnight(now).isoformat()

    completed = db.fetchone(
        "SELECT COUNT(*) AS n FROM tasks "
        "WHERE status = 'completed' AND updated_at >= ?",
        (midnight,),
    )["n"]
    failed = db.fetchone(
        "SELECT COUNT(*) AS n FROM tasks "
        "WHERE status = 'failed' AND updated_at >= ?",
        (midnight,),
    )["n"]
    escalated = db.fetchone(
        "SELECT COUNT(*) AS n FROM tasks "
        "WHERE status = 'blocked' AND block_kind = 'escalated'",
    )["n"]
    # Distinct agents with an unmatched session_start
    active_rows = db.fetchall(
        "SELECT DISTINCT a.agent FROM audit_log a "
        "WHERE a.action = 'session_start' "
        "AND NOT EXISTS ("
        "  SELECT 1 FROM audit_log b "
        "  WHERE b.task_id = a.task_id AND b.agent = a.agent "
        "  AND b.action = 'session_end' AND b.timestamp > a.timestamp"
        ")"
    )
    active_now = len(active_rows)

    return NarrativeCounts(
        completed_today=int(completed),
        failed_today=int(failed),
        escalated_open=int(escalated),
        kb_added_today=kb_store.count_entries_created_since(_local_midnight(now)),
        agents_active_now=active_now,
        spend_today_usd=spend_today(db, now=now),
    )
```

```python
def _tier_for_ratio(failed: int, total: int) -> Literal["ok", "warn", "bad"]:
    if total == 0 or failed == 0:
        return "ok"
    ratio = failed / total
    if ratio < 0.10:
        return "ok"
    if ratio < 0.30:
        return "warn"
    return "bad"


def heartbeat_24h(db, *, now: datetime) -> list[HeartbeatBucket]:
    """24 hourly buckets ending at `now`. Steps = session_start +
    completion_report rows. Tier = ok/warn/bad by failed:total ratio."""
    window_start = (now - timedelta(hours=24)).isoformat()
    rows = db.fetchall(
        "SELECT timestamp, action, payload_json FROM audit_log "
        "WHERE timestamp >= ? AND action IN ('session_start', 'completion_report') "
        "LIMIT 50000",
        (window_start,),
    )
    # Bucket by hour-of-day
    steps_by_hour: dict[int, int] = {h: 0 for h in range(24)}
    failed_by_hour: dict[int, int] = {h: 0 for h in range(24)}
    completion_by_hour: dict[int, int] = {h: 0 for h in range(24)}
    for r in rows:
        ts = datetime.fromisoformat(r["timestamp"])
        h = ts.hour
        steps_by_hour[h] += 1
        if r["action"] == "completion_report":
            completion_by_hour[h] += 1
            import json
            payload = json.loads(r["payload_json"] or "{}")
            if payload.get("status") == "failed":
                failed_by_hour[h] += 1
    return [
        HeartbeatBucket(
            hour=h,
            steps=steps_by_hour[h],
            tier=_tier_for_ratio(failed_by_hour[h], completion_by_hour[h]),
        )
        for h in range(24)
    ]
```

```python
def _verdict_from_payload(action: str, payload_json: str | None) -> Literal["ok", "fail", "warn"] | None:
    import json
    payload = json.loads(payload_json or "{}")
    if action == "completion_report":
        status = payload.get("status")
        if status == "completed":
            return "ok"
        if status == "failed":
            return "fail"
        if status == "blocked":
            return "warn"
    if action == "review_verdict":
        v = payload.get("verdict")
        if v in ("approved", "accept", "ok"):
            return "ok"
        if v in ("request_changes", "rejected"):
            return "fail"
    return None


def recent_activity(db, *, n: int = 6) -> list[ActivityRow]:
    """Last n audit rows of meaningful kinds, DESC by timestamp."""
    placeholders = ",".join("?" * len(_RECENT_ACTIVITY_KINDS))
    rows = db.fetchall(
        f"SELECT timestamp, task_id, agent, action, payload_json "
        f"FROM audit_log WHERE action IN ({placeholders}) "
        f"ORDER BY timestamp DESC LIMIT ?",
        (*_RECENT_ACTIVITY_KINDS, n),
    )
    return [
        ActivityRow(
            timestamp=datetime.fromisoformat(r["timestamp"]),
            who=r["agent"],
            event_kind=r["action"],
            task_id=r["task_id"],
            verdict=_verdict_from_payload(r["action"], r["payload_json"]),
```

```python
def updates_this_week(db, *, now: datetime, kb_store, n: int = 12) -> list[UpdateRow]:
    """Combined feed for the last 7 days: KB entries + learnings promoted."""
    week_start = now - timedelta(days=7)
    items: list[UpdateRow] = []

    # KB entries created this week
    for entry in kb_store.list_entries_created_since(week_start):
        ts = datetime.fromisoformat(entry["created_at"])
        items.append(UpdateRow(
            marker="add",
            text="KB +1",
            meta=entry["slug"],
            timestamp=ts,
        ))

    # Learnings promoted to KB
    rows = db.fetchall(
        "SELECT timestamp, payload_json FROM audit_log "
        "WHERE action = 'learning_promoted' AND timestamp >= ? "
        "ORDER BY timestamp DESC",
        (week_start.isoformat(),),
    )
    import json
    for r in rows:
        payload = json.loads(r["payload_json"] or "{}")
        items.append(UpdateRow(
            marker="info",
            text="Learning promoted to KB",
            meta=payload.get("kb_slug", ""),
            timestamp=datetime.fromisoformat(r["timestamp"]),
        ))

    items.sort(key=lambda x: x.timestamp, reverse=True)
    return items[:n]
```

```python
def escalations_open(db, *, now: datetime) -> list[EscalationRow]:
    """Currently-escalated tasks with the question text from audit payload."""
    rows = db.fetchall(
        "SELECT t.task_id, t.assigned_agent, t.team, t.updated_at, "
        "       a.payload_json AS escalation_payload, a.timestamp AS escalation_ts "
        "FROM tasks t "
        "LEFT JOIN audit_log a ON a.task_id = t.task_id AND a.action = 'escalation' "
        "WHERE t.status = 'blocked' AND t.block_kind = 'escalated' "
        "ORDER BY t.updated_at DESC"
    )
    import json
    result: list[EscalationRow] = []
    for r in rows:
        payload = json.loads(r["escalation_payload"] or "{}")
        question = payload.get("reason") or payload.get("question") or ""
        raised = datetime.fromisoformat(r["escalation_ts"] or r["updated_at"])
        if raised.tzinfo is None:
            raised = raised.replace(tzinfo=timezone.utc)
        moment = now if now.tzinfo else now.replace(tzinfo=timezone.utc)
        age = int((moment - raised).total_seconds())
        result.append(EscalationRow(
            task_id=r["task_id"],
            agent=r["assigned_agent"],
            team=r["team"],
            question=question,
            raised_at=raised,
            age_seconds=max(0, age),
        ))
    return result


def active_by_team(db) -> list[ActiveByTeam]:
    """tasks.status='in_progress' grouped by team."""
    rows = db.fetchall(
        "SELECT team, task_id FROM tasks WHERE status = 'in_progress' "
        "ORDER BY team, updated_at DESC"
    )
    groups: dict[str, list[str]] = {}
    for r in rows:
        groups.setdefault(r["team"], []).append(r["task_id"])
    return [
        ActiveByTeam(team=team, count=len(ids), task_ids=ids[:10])
        for team, ids in sorted(groups.items())
    ]
```

```python
def _acceptance_pct_for_window(
    db, *, team_id: str, start: datetime, end: datetime,
) -> tuple[int, int]:
    """Return (approved_count, total_count) of review_verdict rows in window."""
    rows = db.fetchall(
        "SELECT a.payload_json FROM audit_log a "
        "JOIN tasks t ON t.task_id = a.task_id "
        "WHERE a.action = 'review_verdict' "
        "  AND t.team = ? "
        "  AND a.timestamp >= ? AND a.timestamp < ?",
        (team_id, start.isoformat(), end.isoformat()),
    )
    import json
    approved = 0
    total = len(rows)
    for r in rows:
        payload = json.loads(r["payload_json"] or "{}")
        v = payload.get("verdict")
        if v in ("approved", "accept", "ok"):
            approved += 1
    return approved, total


def org_pulse_7d(db, *, now: datetime, teams) -> list[TeamPulse]:
    """Per-team 7d acceptance + trend + 12-week sparkline."""
    week_now_start = now - timedelta(days=7)
    week_prior_start = now - timedelta(days=14)

    result: list[TeamPulse] = []
    for team in teams.list_teams():
        # Current 7d
        approved_now, total_now = _acceptance_pct_for_window(
            db, team_id=team.team_id, start=week_now_start, end=now,
        )
        # Prior 7d
        approved_prior, total_prior = _acceptance_pct_for_window(
            db, team_id=team.team_id, start=week_prior_start, end=week_now_start,
        )
        acceptance = int(round(100 * approved_now / total_now)) if total_now else 0
        prior_acc = int(round(100 * approved_prior / total_prior)) if total_prior else 0
        trend_delta = acceptance - prior_acc

        # 12-week sparkline (oldest first), each value in [0, 1]
        sparkline: list[float] = []
        for i in range(12, 0, -1):
            w_start = now - timedelta(days=i * 7)
            w_end = now - timedelta(days=(i - 1) * 7)
            approved, total = _acceptance_pct_for_window(
                db, team_id=team.team_id, start=w_start, end=w_end,
            )
            sparkline.append(approved / total if total else 0.0)

        result.append(TeamPulse(
            team=team.team_id,
            acceptance_pct=acceptance,
            trend_delta=trend_delta,
            sparkline=sparkline,
            members=len(team.members),
            lead=team.lead,
        ))
    return result
```

```typescript
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { getDashboardSummary } from '@/lib/api/dashboard';
import { request } from '@/lib/api/client';
import type { DashboardApi } from './DataContext';

export function makeRealDashboard(orgSlug: string): DashboardApi {
  return {
    useDashboardSummary: () => {
      const q = useQuery({
        queryKey: ['dashboard', 'summary', orgSlug],
        queryFn: () => getDashboardSummary(orgSlug),
        staleTime: 30_000,
      });
      return {
        data: q.data,
        isLoading: q.isLoading,
        isError: q.isError,
        error: q.error ?? undefined,
        refetch: () => {
          void q.refetch();
        },
      };
    },
    useResolveEscalation: () => {
      const qc = useQueryClient();
      const m = useMutation({
        mutationFn: async (args: {
          taskId: string;
          reply: string;
          promoteToKb?: { slug: string; type: string };
        }) => {
          await request(
            `/api/v1/orgs/${orgSlug}/tasks/${args.taskId}/resolve-escalation`,
            { method: 'POST', body: JSON.stringify({ reply: args.reply }) },
          );
          if (args.promoteToKb) {
            await request(`/api/v1/orgs/${orgSlug}/kb/entries`, {
              method: 'POST',
              body: JSON.stringify({
                slug: args.promoteToKb.slug,
                type: args.promoteToKb.type,
                body: args.reply,
                source_task: args.taskId,
              }),
            });
          }
          await qc.invalidateQueries({ queryKey: ['dashboard', 'summary', orgSlug] });
        },
      });
      return { mutate: m.mutateAsync, isPending: m.isPending };
    },
  };
}
```

```typescript
import type { JSX } from 'react';
import { cn } from '@/lib/utils';

type SparklineVariant = 'default' | 'green' | 'yellow' | 'red';

interface SparklineProps {
  data: number[];
  width?: number;
  height?: number;
  variant?: SparklineVariant;
}

const VARIANT_STROKE: Record<SparklineVariant, string> = {
  default: 'stroke-text-secondary',
  green: 'stroke-tier-green',
  yellow: 'stroke-tier-yellow',
  red: 'stroke-tier-red',
};

export function Sparkline({
  data,
  width = 64,
  height = 16,
  variant = 'default',
}: SparklineProps): JSX.Element {
  if (data.length === 0) {
    return <svg width={width} height={height} aria-hidden="true" />;
  }
  const minY = Math.min(...data, 0);
  const maxY = Math.max(...data, 1);
  const range = maxY - minY || 1;
  const stepX = data.length === 1 ? 0 : width / (data.length - 1);
  const points = data
    .map((v, i) => {
      const x = i * stepX;
      const y = height - ((v - minY) / range) * height;
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    })
    .join(' ');

  return (
    <svg width={width} height={height} aria-hidden="true">
      <polyline
        points={points}
        fill="none"
        strokeWidth={1.5}
        className={cn('stroke-current', VARIANT_STROKE[variant])}
      />
    </svg>
  );
}

export const meta = {
  name: 'Sparkline',
  layer: 'pattern',
  import: '@/design-system/patterns/Sparkline',
  variants: { variant: ['default', 'green', 'yellow', 'red'] },
  consumes: ['typography.mono_sm', 'colors.semantic.dark.tier'],
  example: '<Sparkline data={[0.8, 0.84, 0.78, 0.82, 0.86]} />',
} as const;
```

```typescript
}: EscalationInboxRowProps): JSX.Element {
  const [reply, setReply] = useState('');
  const [promote, setPromote] = useState(true);
  const taRef = useRef<HTMLTextAreaElement>(null);
  const { mutate, isPending } = useResolveEscalation();

  useEffect(() => {
    if (expanded) {
      const t = setTimeout(() => taRef.current?.focus(), 30);
      return () => clearTimeout(t);
    }
  }, [expanded]);

  async function submit() {
    if (!reply.trim()) return;
    await mutate({
      taskId: row.task_id,
      reply,
      promoteToKb: promote
        ? { slug: `escalation-${row.task_id.toLowerCase()}`, type: 'ruling' }
        : undefined,
    });
  }

  function onKeyDown(e: React.KeyboardEvent<HTMLTextAreaElement>) {
    if (e.key === 'Escape') {
      onCollapse();
      e.preventDefault();
    }
    if (e.key === 'Enter' && (e.metaKey || e.ctrlKey)) {
      void submit();
      e.preventDefault();
    }
  }

  return (
    <div
      className="border-border-subtle rounded-md border p-3"
      onClick={() => !expanded && onExpand()}
    >
      <div className="flex items-baseline gap-2">
        <span className="text-text-primary font-mono text-xs font-medium">
          {row.agent}
        </span>
        <span className="text-text-muted font-mono text-xs">·</span>
        <span className="text-text-muted font-mono text-xs">
          {row.task_id} · {row.team} · {relativeAge(row.age_seconds)}
        </span>
      </div>
      <p className="text-text-primary mt-1 text-sm">{row.question}</p>

      {expanded && (
        <div className="mt-3 space-y-2" onClick={(e) => e.stopPropagation()}>
          <textarea
            ref={taRef}
            value={reply}
            onChange={(e) => setReply(e.target.value)}
            onKeyDown={onKeyDown}
            placeholder={`Your response to ${row.agent}…`}
            className="border-border-default bg-surface-canvas text-text-primary w-full rounded border p-2 text-sm"
            rows={3}
          />
          <label className="text-text-secondary flex items-center gap-2 text-xs">
            <input
              type="checkbox"
              checked={promote}
              onChange={(e) => setPromote(e.target.checked)}
            />
            Save reply as KB ruling
          </label>
          <div className="flex justify-end gap-2">
            <Button variant="ghost" onClick={onCollapse} type="button">
              Cancel
            </Button>
            <Button
              variant="primary"
              onClick={() => void submit()}
              disabled={isPending || !reply.trim()}
              type="button"
            >
              {promote ? 'Reply & promote to KB' : 'Send reply'}
            </Button>
          </div>
        </div>
      )}
    </div>
  );
}
```
